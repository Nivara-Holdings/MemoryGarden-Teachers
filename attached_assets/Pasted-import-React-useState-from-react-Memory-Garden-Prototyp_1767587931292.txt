        <div style={headerStyle}>
          <span style={{ width: '24px' }}></span>
          <span style={{ fontSize: '18px' }}>🌱</span>
          <span 
            onClick={() => { setIsChildMode(false); setCurrentScreen('welcome'); }}
            style={{ fontSize: '20px', cursor: 'pointer', color: colors.warmGray }}
          >✕</span>
        </div>
        
        <div style={{ padding: '24px' }}>
          <div style={{ textAlign: 'center', marginBottom: '32px' }}>
            <h1 style={{ fontSize: '28px', fontWeight: '400', marginBottom: '8px' }}>
              Your Garden 🌱
            </h1>
            <p style={{ color: colors.warmGray, fontSize: '15px' }}>
              {sharedMemories.length} memories from people who love you
            </p>
          </div>
          
          {sharedMemories.map((memory) => {
            const typeStyle = getTypeColor(memory.type);
            
            // Voice memo card for child view
            if (memory.type === 'voiceMemo') {
              return (
                <div 
                  key={memory.id} 
                  style={{
                    ...cardStyle,
                    backgroundColor: '#FFFFFF',
                    borderRadius: '20px',
                    padding: '24px',
                    borderLeft: `4px solid ${typeStyle.accent}`
                  }}
                >
                  <VoiceMemoPlayer memory={memory} isChild={true} />
                  <div style={{ 
                    display: 'flex', 
                    justifyContent: 'flex-end',
                    marginTop: '8px'
                  }}>
                    <span style={{ fontSize: '13px', color: colors.warmGray }}>{memory.date}</span>
                  </div>
                </div>
              );
            }
            
            return (
              <div 
                key={memory.id} 
                style={{
                  ...cardStyle,
                  backgroundColor: '#FFFFFF',
                  borderRadius: '20px',
                  padding: '24px',
                  borderLeft: `4px solid ${typeStyle.accent}`
                }}
              >
                <div style={{
                  display: 'inline-flex',
                  alignItems: 'center',
                  gap: '6px',
                  backgroundColor: typeStyle.bg,
                  padding: '4px 12px',
                  borderRadius: '12px',
                  fontSize: '12px',
                  color: typeStyle.accent,
                  marginBottom: '12px'
                }}>
                  {typeStyle.emoji} {memory.type === 'fromOthers' ? `From ${memory.from}` : getTypeLabel(memory.type)}
                </div>
                
                {memory.hasPhoto && (
                  <div style={{ 
                    height: '180px', 
                    backgroundColor: typeStyle.bg, 
                    borderRadius: '16px', 
                    marginBottom: '20px',
                    display: 'flex',
                    alignItems: 'center',
                    justifyContent: 'center'
                  }}>
                    <span style={{ fontSize: '40px' }}>
                      {memory.type === 'keepsake' ? '🖼️' : '📷'}
                    </span>
                  </div>
                )}
                
                <p style={{ fontSize: '18px', lineHeight: '1.8', fontStyle: 'italic', color: colors.softBlack }}>
                  "{memory.note}"
                </p>
                
                <div style={{ 
                  display: 'flex', 
                  justifyContent: 'space-between', 
                  alignItems: 'center',
                  marginTop: '16px',
                  paddingTop: '16px',
                  borderTop: `1px solid ${colors.border}`
                }}>
                  <div>
                    <p style={{ fontSize: '14px', color: colors.softBlack, fontWeight: '500' }}>
                      {memory.from}
                    </p>
                    <p style={{ fontSize: '12px', color: colors.warmGray }}>{memory.date}</p>
                  </div>
                  <span style={{ fontSize: '24px' }}>{typeStyle.emoji}</span>
                </div>
              </div>
            );
          })}
          
          <div style={{ 
            textAlign: 'center', 
            padding: '40px 20px',
            color: colors.warmGray
          }}>
            <p style={{ fontSize: '15px', fontStyle: 'italic', lineHeight: '1.7' }}>
              This is your garden of memories.<br/>
              Each one planted with love by people<br/>
              who see the best in you.
            </p>
          </div>
        </div>
      </div>
    );
  };

  // Render current screen
  const renderScreen = () => {
    switch(currentScreen) {
      case 'welcome': return <WelcomeScreen />;
      case 'signup': return <SignUpScreen />;
      case 'addChild': return <AddChildScreen />;
      case 'home': return <HomeScreen />;
      case 'chooseType': return <ChooseTypeScreen />;
      case 'addMemory': return <AddMemoryScreen />;
      case 'recordVoice': return <RecordVoiceScreen />;
      case 'reviewVoice': return <ReviewVoiceScreen />;
      case 'suggestions': return <SuggestionsScreen />;
      case 'review': return <ReviewScreen />;
      case 'childView': return <ChildViewScreen />;
      default: return <WelcomeScreen />;
    }
  };

  return (
    <div style={{ backgroundColor: '#F5F5F5', minHeight: '100vh' }}>
      {renderScreen()}
    </div>
  );
};

export default MemoryGardenApp;
